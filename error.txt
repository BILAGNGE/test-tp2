# Nombre d'utilisateurs
Write-Host "Utilisateurs :" -ForegroundColor Cyan
(Get-ADUser -Filter * -SearchBase "OU=Utilisateurs,OU=LaPlateforme,DC=laplateforme,DC=io").Count

# Liste des groupes
Write-Host "`nGroupes :" -ForegroundColor Cyan
Get-ADGroup -Filter * -SearchBase "OU=Groupes,OU=LaPlateforme,DC=laplateforme,DC=io" | Select-Object Name

# Vérification d'un user (exemple THILLOT)
Write-Host "`nGroupes de THILLOT :" -ForegroundColor Cyan
(Get-ADUser -Identity "mthillot" -Properties MemberOf).MemberOf | ForEach-Object { ($_ -split ',')[0] -replace 'CN=' }
