$groupesManquants = @("Médical", "Hébergement", "cadre de santé", "Maîtresse de Maison", "Médecin", "Secrétaire")

foreach ($grp in $groupesManquants) {
    $existing = Get-ADGroup -Filter * -SearchBase "OU=Groupes,OU=LaPlateforme,DC=laplateforme,DC=io" | Where-Object { $_.Name -like "*$($grp.Substring(0,3))*" }
    if ($existing) {
        Write-Host "Suppression ancien : $($existing.Name)" -ForegroundColor Red
        Remove-ADGroup -Identity $existing -Confirm:$false
    }
    New-ADGroup -Name $grp -GroupScope Global -GroupCategory Security -Path "OU=Groupes,OU=LaPlateforme,DC=laplateforme,DC=io"
    Write-Host "Créé : $grp" -ForegroundColor Green
}
